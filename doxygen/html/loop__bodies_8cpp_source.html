25 {
26  for (int i = range.begin(); i != range.end(); ++i)
27  {
28  // Determine the length of the line by computing the difference between the
29  // current LF character and the previous (does always work, since the first element
30  // in the "lfs" vector is a "-1").
31  // Then, allocate an appropriate amount of memory (plus 1 byte for the trailing 0-byte).
32  char* b = (char*) malloc(lfs->at(i) - lfs->at(i - 1) + 1);
33 
34  // Copy the line into the previously allocated line buffer.
35  strncpy(b, (input + lfs->at(i - 1) + 1), lfs->at(i) - lfs->at(i - 1) - 1);
36 
37  // Add 0-byte to mark string end.
38  b[lfs->at(i) - lfs->at(i - 1) - 1] = 0x00;
39

211 {

213 
214  // First, if a direct flight exist, it must be in the final travels.
215  // Should not occur, since we already filter these out in fill_travel.
216  if (current_city->to == to)
