312 time_t timegm(struct tm *tm)
313 {
314  // Create a simple hash from the year and month.
315  int year = tm->tm_year * 100 + tm->tm_mon;
316  time_t month_ts;
317 
318  // Try to look up the month/year hash in the time map. Use the existing
319  // base timestamp if it exists, otherwise compute it.
320  concurrent_hash_map<int, time_t>::const_accessor a;
321  if (times.find(a, year))
322  {
323  month_ts = a->second;
324  a.release();
325  }
326  else
327  {
328  concurrent_hash_map<int, time_t>::accessor b;
329  if (times.insert(b, year))
330  {
331  int years = tm->tm_year > 100 ? tm->tm_year - 70 : tm->tm_year;
332 
333  // Build offset for year (not yet considering leap years).
334  unsigned long offset = years * 31536000; // = 365 * 24 * 60 * 60
335 
336  // Add additional days to offset for each passed leap year.
337  for (int y = years; y >= 0; y--)
338  {
339  if (((y + 1970) % 4 == 0) && ((y + 1970) % 100 != 0))
340  {
341  offset += 86400;
342  }
343  }
344 
345  // Add additional days to offset for each passed month. Have to consider
346  // different month lengths and leap years.
347  for (int m = 0; m < tm->tm_mon; m++)
348  {
349  bool is_leap_year = (tm->tm_year + 1900 % 4 == 0)
350  && (tm->tm_year + 1900 % 100 != 0);
351 
352  if (m == 1) offset += (is_leap_year ? 29 : 28) * 86400;
353  else if ((m < 7 && m % 2 == 0) || (m >= 7 && m % 2 == 1)) offset += 31
354  * 86400;
355  else offset += 30 * 86400;
356  }
357 
358  if (tm->tm_mon != 1) offset += 86400;
359 
360  b->second = offset;
361  month_ts = offset;
362  }
363  else
364  {
365  month_ts = b->second;
366  }
367  }
368 
369  month_ts += (tm->tm_mday - 1) * 86400 + tm->tm_hour * 3600 + tm->tm_min * 60
370  + tm->tm_sec;
371  return month_ts;
372 }

637

639  // Exit with an error message if either operation fails.
640  fd = open(filename.c_str(), O_RDONLY);
641  if (fd && fstat(fd, &stat) != 0)
642  {
643  cerr << "Could not open or stat " << filename << endl;
644  exit(130);
645  }
646 
647  // Get file size from stat call and map the entire file into memory.
648  l = stat.st_size;
649  m = (char*) mmap((void*) m, l, PROT_READ, MAP_PRIVATE, fd, 0);
650 
651  // Iterate through the entire file and search for line feeds.
652  // Store all linefeeds into a vector.
653  vector<int> lfs;
654  lfs.push_back(-1);
655  for (off_t i = 38; i < l; i++)
656  {
657  // 10 = LF
658  if (i < l && m[i] == 10)
659  {
660  lfs.push_back(i);
661 
662  // Each line is at least 39 characters long (assuming 14 character
663  // datetime values and at least 1 character for every other value),
664  // so it is safe to assume that no linefeed will occur for the next
665  // 39 characters.
666  i += 39;
667  }
668  }
669

671  ParseFlightsLoop pfl(m, &lfs, parameters);
672  parallel_for(blocked_range<int>(1, lfs.size()), pfl);
673 
674  // Unmap file from memory and close file handle.
675  munmap(m, stat.st_size);
